algoritmo "AnalizadorValores"
var
   v, c, s, div5, nulo, spares: Inteiro
   media: Real
inicio
      s <- 0
      div5 <- 0
      nulo <- 0
      spares <- 0
      Para c <- 1 ate 5 faca
      Escreva ("Digite o ", c, "o. valor: ")
      Leia (v)
      s <- s + v
        Se (v%5=0) entao
           div5 <- div5 + 1
        FimSe
        Se (v=0) entao
           nulo <- nulo + 1
        FimSe
        Se (v%2=0) entao
           spares <- spares + v
        FimSe
      FimPara
      media <- s / (c-1)
      EscrevaL ("A soma entre os valores são de ", s)
      EscrevaL ("Valores divisíveis por cinco: ", div5)
      EscrevaL ("Valores nulos: ", nulo)
      EscrevaL ("A soma dos valores pares são ", spares)
      EscrevaL ("A média entre os valores são ", media:4:1)
fimalgoritmo
